"Find all pizzerias that serve at least one pizza for less than $10 that both Amy and Fay eat."

\project_{pizzeria} (
    (\select_{name="Amy" or name="Fay"} (
        \select_{price<10} Serves \join Eats
    ))
    \diff
    (\select_{name="Amy" and name!="Fay"} (
        \select_{price<10} Serves \join Eats
    ))
)
